import java.util.Scanner;

public class Main{
    
    public static void main(String[] args){
        String st = "";
        Scanner sc = new Scanner(System.in);
        char[] ch = sc.nextLine().toCharArray();
        for(int i = ch.length-1 ; i>=0 ; i--){
          st += ch[i];  
        }
        System.out.println(st);
    }
}
